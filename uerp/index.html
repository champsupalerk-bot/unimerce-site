async function loadUerpHeader() { try { const url = './uerpheader.html'; console.log("Loading header:", url); const res = await fetch(url); console.log( "Header status:", res.status ); if(!res.ok){ throw new Error( "Header not found: " + res.status ); } const html = await res.text(); document .getElementById('uerp-header-container') .innerHTML = html; } catch(e) { console.error( "UERP HEADER ERROR:", e ); } } loadUerpHeader();
